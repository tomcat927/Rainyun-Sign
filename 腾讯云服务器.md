
二、先执行这几条命令确认实际路径
你直接复制执行：

<BASH>
which chromium
which chromium-browser
which google-chrome
which google-chrome-stable
which chromedriver
ls -l /usr/bin/chromium /usr/bin/chromium-browser /usr/bin/google-chrome /usr/bin/google-chrome-stable /usr/bin/chromedriver 2>/dev/null



# 腾讯云 的 chromedriver、chrome-headless-shell驱动位置

root@VM-20-8-debian:~# which chromium
root@VM-20-8-debian:~# which chromium-browser
root@VM-20-8-debian:~# which chromedriver
/usr/local/bin/chromedriver
root@VM-20-8-debian:~# which chrome-headless-shell
/usr/local/bin/chrome-headless-shell
root@VM-20-8-debian:~#


浏览器：
/usr/local/bin/chrome-headless-shell
驱动：
/usr/local/bin/chromedriver

先确认这两个能不能正常执行
先执行：

<BASH>
/usr/local/bin/chrome-headless-shell --version
/usr/local/bin/chromedriver --version
如果都能输出版本号，说明可以继续改代码。

root@VM-20-8-debian:~# /usr/local/bin/chrome-headless-shell --version
Google Chrome for Testing 139.0.7258.68
root@VM-20-8-debian:~# /usr/local/bin/chromedriver --version
ChromeDriver 139.0.7258.66 (a62d329947691f76c376a873eae39f56381103c8-refs/branch-heads/7258@{#2226})
root@VM-20-8-debian:~#
